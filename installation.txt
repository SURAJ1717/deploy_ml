After login to AWS EC2

$ sudo apt-get update
$ sudo apt-get install apache2 libapache2-mod-wsgi python-pip

cd /var/www/
$ git clone https://github.com/SURAJ1717/deploy_ml.git

cd /var/www/deploy_ml
$ sudo pip install flask virtualenv
$ virtualenv venv
$ source venv/bin/activate

$ sudo pip install -r requirments.txt
$ sudo pip install astroid gunicorn kiwisolver matplotlib numpy pandas pylint scikit-learn scipy seaborn
$ sudo pip3 install typed-ast xgboost

$ python app.py

// Apache2 configuration
cd /etc/apache2/sites-available
$ sudo nano flaskapp.conf
    
    <VirtualHost *:80>
        ServerAdmin webmaster@localhost
        DocumentRoot /var/www/

        WSGIDaemonProcess flaskapp threads=5
        WSGIScriptAlias / /var/www/deploy_ml/flaskapp.wsgi

        <Directory deploy_ml>
            WSGIProcessGroup flaskapp
            WSGIApplicationGroup %{GLOBAL}
            Order deny,allow
            Allow from all
        </Directory>

        ErrorLog ${APACHE_LOG_DIR}/error.log
        CustomLog ${APACHE_LOG_DIR}/access.log combined
    </VirtualHost>

$ a2ensites flaskapp.conf
$ sudo service apache2 restart
